
//PCI-POP
SELECT 
    STATE_NAME, PCI_2010_11,SUM(POP_2011)
FROM
    POP_DISTRICTWISE PD, MAIN_TABLE MT, PCI_STATEWISE PS
WHERE
    PD.STATE_ID=MT.STATE_ID AND PS.STATE_ID=MT.STATE_ID AND PD.STATE_ID=PS.STATE_ID AND PD.DIST_ID=MT.DIST_ID
GROUP BY
    MT.STATE_ID, PD.STATE_ID 
	
//PCI-LIT	
SELECT 
    STATE_NAME, PCI_2010_11,LIT_2011
FROM
    LITERACY_STATEWISE LS, MAIN_TABLE MT, PCI_STATEWISE PS
WHERE
    LS.STATE_ID=MT.STATE_ID AND PS.STATE_ID=MT.STATE_ID AND LS.STATE_ID=PS.STATE_ID
GROUP BY
    MT.STATE_ID;
	

//HH-URT	
SELECT 
    STATE_NAME,URT_ID,HHSIZE_1+HHSIZE_2+HHSIZE_3+ HHSIZE_4+ HHSIZE_5+ HHSIZE_6+ HHSIZE_7_10+ HHSIZE_11_14+ HHSIZE_15_MORE as HH_NO
FROM
    HH, MAIN_TABLE MT
WHERE
    HH.STATE_ID=MT.STATE_ID
group by
    URT_ID,MT.STATE_ID
	
	
//GER-LIT
SELECT 
    STATE_NAME, ALL_TOTAL, LIT_2011
FROM
    LITERACY_STATEWISE LS, MAIN_TABLE MT, GROSS_ENROLL_RATIO GER
WHERE
    LS.STATE_ID=MT.STATE_ID AND GER.STATE_ID=MT.STATE_ID AND LS.STATE_ID=GER.STATE_ID
GROUP BY
    MT.STATE_ID;
	
	
//GDP-PCI
SELECT 
    STATE_NAME,SUM(GDP_2010_11),PCI_2010_11
FROM
    PCI_STATEWISE PS, MAIN_TABLE MT, GDP_DISTRICTWISE GD
WHERE
    PS.STATE_ID=MT.STATE_ID AND GD.STATE_ID=MT.STATE_ID AND MT.DIST_ID=GD.DIST_ID
GROUP BY
    GD.STATE_ID,MT.STATE_ID;
	
	
	
//GDP-LIT
SELECT 
    STATE_NAME,SUM(GDP_2010_11),LIT_2011
FROM
    LITERACY_STATEWISE LS, MAIN_TABLE MT, GDP_DISTRICTWISE GD
WHERE
    LS.STATE_ID=MT.STATE_ID AND GD.STATE_ID=MT.STATE_ID AND MT.DIST_ID=GD.DIST_ID
GROUP BY
    GD.STATE_ID,MT.STATE_ID;
